#!/bin/bash

# Set up script for ScarletRice1
# By: LBRM
# Run the script as root user


# Switch to root user and execute commands
# su - 

# Update system and upgrade existing packages
sudo apt update && sudo apt upgrade -y


############## INSTALLATIONS ################

# Install Build Tools
sudo apt install -y build-essential pkg-config cmake

# Development headers for x11 libraries
sudo apt install -y libx11-dev libxft-dev libxinerama-dev

# Install Core Utilities
sudo apt install -y git wget curl xclip unzip checkinstall

# Install GTK Toolkit
sudo apt install -y libgtk-3-dev libgtk2.0-dev

# Install Xorg
sudo apt install -y xorg

# Install Compositor for x11
sudo apt install -y picom

# Install Image Viewer for x11
sudo apt install -y feh

# Install AMD graphics drivers and necessary firmware * for Asus rx 580
# Remove or change this line if installing on a different GPU
sudo apt install -y mesa-utils xserver-xorg-video-amdgpu firmware-amd-graphics

# Install Monitoring Tool
sudo apt install -y lm-sensor sysstat htop 

# Install File Navigation
sudo apt install -y ranger

# Install Blutooth Utilities
sudo apt install -y bluetooth bluez

# Install Wifi Utilities
sudo apt install -y network-manager

# Install Text Editor
sudo apt install -y vim

# Install Audio Utilities
sudo apt install -y pulseaudio

# Install Terminal Emulator (Alacritty)
sudo apt install -y alacritty

# Install Terminal Multiplexer (tmux)
sudo apt install -y tmux

# Install Window Manager (i3)
sudo apt install -y i3 i3status i3lock dmenu

# Install Desktop Manger (lightdm)
sudo apt install -y lightdm lightdm-gtk-greeter

# Install Desktop Applications
sudo apt install -y rofi polybar

# Install Image Editor (Inkscape, GIMP)
sudo apt install -y gimp inkscape

# Install GitHub CLI
curl -fsSL \
  https://cli.github.com/packages/githubcli-archive-keyring.gpg \
  | sudo gpg --dearmor -o /usr/share/keyrings/githubcli-archive-keyring.gpg
echo \
  "deb [arch=amd64 signed-by=/usr/share/keyrings/githubcli-archive-keyring.gpg] \
  https://cli.github.com/packages stable main" \
  | sudo tee /etc/apt/sources.list.d/github-cli.list
sudo apt update
sudo apt install -y gh

# Install Web Browser (Brave)
sudo curl -fsSLo /usr/share/keyrings/brave-browser-archive-keyring.gpg \
  https://brave-browser-apt-release.s3.brave.com/brave-browser-archive-keyring.gpg
echo "deb [signed-by=/usr/share/keyrings/brave-browser-archive-keyring.gpg] \
  https://brave-browser-apt-release.s3.brave.com/ stable main" \
  | sudo tee /etc/apt/sources.list.d/brave-browser-release.list
sudo apt update
sudo apt install -y brave-browser

# Install Text Editor (nvim)
# Download the Neovim AppImage
curl -L -o nvim.appimage \
  https://github.com/neovim/neovim/releases/download/stable/nvim.appimage
# Make the AppImage executable
chmod +x nvim.appimage
# Move the AppImage to a directory in PATH (e.g., /usr/local/bin)
sudo mv nvim.appimage /usr/local/bin/nvim
# Install ripgrep for telescope
sudo apt install -y ripgrep

# Install WebI package manager for Web Development
curl -sS https://webinstall.dev/webi | bash
